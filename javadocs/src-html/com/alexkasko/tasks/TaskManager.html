001    package com.alexkasko.tasks;
002    
003    import java.util.Collection;
004    
005    /**
006     * DAO interface for tasks
007     *
008     * @author alexkasko
009     * Date: 5/17/12
010     * @see TaskEngine
011     * @see Task
012     */
013    public interface TaskManager<T extends Task> {
014        /**
015         * This method is used to load new and resumed tasks to run.
016         * All returning tasks should be switched into 'processing' status in DB.
017         * It's implementation responsibility to not return tasks that may be running in that moment.
018         *
019         * @return collection of tasks to run
020         */
021        Collection<? extends T> markProcessingAndLoad();
022    
023        /**
024         * This method will be used to cache all suspended task ids on engine startup.
025         *
026         * @return collection of suspended tasks ids
027         */
028        Collection<Long> loadSuspendedIds();
029    
030        /**
031         * Changes task stage, will be called between stages processing
032         *
033         * @param taskId task id
034         * @param stage new stage
035         */
036        void updateStage(long taskId, String stage);
037    
038        /**
039         * Changes task status to default, will be called after successful processing of last stage
040         *
041         * @param taskId task id
042         */
043        void updateStatusDefault(long taskId);
044    
045        /**
046         * Changes task status to default, will be called after on task suspend
047         *
048         * @param taskId task id
049         */
050        void updateStatusSuspended(long taskId);
051    
052        /**
053         * Changes task status into "error" and task stage into last completed stage
054         *
055         * @param taskId task id
056         * @param e exception
057         * @param lastCompletedStage name of last completed stage
058         */
059        void updateStatusError(long taskId, Exception e, String lastCompletedStage);
060    }
