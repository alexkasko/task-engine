001    package com.alexkasko.tasks;
002    
003    /**
004     * Interface for separate units of business oriented work on task
005     *
006     * @author alexkasko
007     * Date: 5/17/12
008     * @see TaskEngine
009     * @see TaskEngine#checkSuspended(long)
010     * @see Task
011     * @see TaskProcessorProvider
012     */
013    public interface TaskStageProcessor {
014    
015        /**
016         * Some business oriented work to do on task.
017         * {@link TaskEngine#checkSuspended(long)} method should be called periodically,
018         * {@link TaskSuspendedException} must be thrown on successful suspension check
019         *
020         * @param taskId task id
021         * @throws TaskSuspendedException task will rolled back to last completed stage and switched into 'suspended' status
022         * @throws Exception task will rolled back to last completed stage and switched into 'error' status
023         */
024        void process(long taskId) throws Exception;
025    }
