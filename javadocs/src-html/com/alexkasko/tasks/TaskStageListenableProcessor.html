001    package com.alexkasko.tasks;
002    
003    import java.util.List;
004    
005    /**
006     * {@link TaskStageProcessor} extension to allow additional operations execution
007     * before/after task stage processing.
008     * Listeners are separated from task processors to (among others)
009     * simplify their declarative transaction management.
010     *
011     * @author alexkasko
012     * Date: 11/11/12
013     * @see TaskEngine
014     * @see TaskStageListener
015     */
016    public interface TaskStageListenableProcessor extends TaskStageProcessor {
017        /**
018         * List of additional operations, that will be called by {@link TaskEngine} before stage processing
019         *
020         * @return before listeners
021         */
022        List<? extends TaskStageListener> beforeStartListeners();
023    
024        /**
025         * List of additional operations, that will be called by {@link TaskEngine} after stage processing
026         *
027         * @return after listeners
028         */
029        List<? extends TaskStageListener> afterFinishListeners();
030    }
