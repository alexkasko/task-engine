001    package com.alexkasko.tasks;
002    
003    import java.io.Serializable;
004    import java.util.*;
005    
006    /**
007     * Implementation of task stages list. Must be provided by {@link Task} instances. Thread-safe.
008     *
009     * @author alexkasko
010     * Date: 5/22/12
011     * @see TaskEngine
012     * @see Task
013     * @see TaskProcessorProvider
014     */
015    public class TaskStageChain implements Serializable {
016        private static final long serialVersionUID = 7486673100573364684L;
017        private final Map<String, Stage> stageMap;
018        private final List<Stage> stageList;
019    
020        private TaskStageChain(List<Stage> stageList) {
021            this.stageList = stageList;
022            this.stageMap = new LinkedHashMap<String, Stage>();
023            for(Stage ts : stageList) {
024                if(!ts.isStart()) stageMap.put(ts.getIntermediate(), ts);
025                stageMap.put(ts.getCompleted(), ts);
026            }
027        }
028    
029        /**
030         * @param stage stage name
031         * @return {@link Stage} instance for given name
032         */
033        public Stage forName(String stage) {
034            if(null == stage) throw new IllegalArgumentException("Null stage provided");
035            Stage res = stageMap.get(stage);
036            if(null == res) throw new IllegalArgumentException(
037                    "Unknown stage provided, valid stages are: '" + stageList + "'");
038            return res;
039        }
040    
041        /**
042         * @param stage input stage
043         * @return previous stage
044         */
045        public Stage previous(Stage stage) {
046            int ind = index(stage);
047            if(0 == ind) throw new IllegalArgumentException(
048                    "Start stage: '" + stage + "', has no previous stage, valid stages are: '" + stageList + "'");
049            return stageList.get(ind - 1);
050        }
051    
052        /**
053         * @param stage input stage
054         * @return next stage
055         */
056        public Stage next(Stage stage) {
057            int ind = index(stage);
058            if(stageList.size() - 1 == ind) throw new IllegalArgumentException(
059                    "End stage: '" + stage + "', has no next stage, valid stages are: '" + stageList + "'");
060            return stageList.get(ind + 1);
061        }
062    
063        /**
064         * @param stage input stage
065         * @return whether next stage exists for given stage
066         */
067        public boolean hasNext(Stage stage) {
068            return stageList.size() - 1 > index(stage);
069        }
070    
071        /**
072         * @param startStage start stage
073         * @return {@link Builder} builder for chain
074         */
075        public static Builder builder(Enum<?> startStage) {
076            if(null == startStage) throw new IllegalArgumentException("Null startStage provided");
077            return builder(startStage.name());
078        }
079    
080        /**
081         * @param startStage start stage name
082         * @return {@link Builder} builder for chain
083         */
084        public static Builder builder(String startStage) {
085            if(null == startStage) throw new IllegalArgumentException("Null startStage provided");
086            return new Builder(startStage);
087        }
088    
089        private int index(Stage stage) {
090            if(null == stage) throw new IllegalArgumentException("Null stage provided");
091            // for short lists this should be faster on list than on set
092            int pos = stageList.indexOf(stage);
093            if(-1 == pos) throw new IllegalArgumentException("Unknown stage provided, valid stages are: '" + stageList + "'");
094            return pos;
095        }
096    
097    
098        /**
099         * Builder class for {@link TaskStageChain}, not thread-safe
100         */
101        public static class Builder {
102            private final List<Stage> list = new ArrayList<Stage>();
103            private final Set<String> stages = new HashSet<String>();
104    
105            private Builder(String startStage) {
106                this.list.add(new Stage(startStage));
107                this.stages.add(startStage);
108            }
109    
110            /**
111             * Adds new enum stage to chain
112             *
113             * @param intermediate intermediate stage, e.g. 'running', 'loading_data'
114             * @param completed completed stage, e.g. 'finished', 'data_loaded'
115             * @param processorId id of the processor that will be used for this stage
116             * @return builder instance
117             */
118            public Builder add(Enum<?> intermediate, Enum<?> completed, String processorId) {
119                if(null == intermediate) throw new IllegalArgumentException("Null intermediate stage provided");
120                if(null == completed) throw new IllegalArgumentException("Null completed stage provided");
121                return add(intermediate.name(), completed.name(), processorId);
122            }
123    
124            /**
125             * Adds new stage to chain
126             *
127             * @param intermediate intermediate stage name, e.g. 'running', 'loading_data'
128             * @param completed completed stage name, e.g. 'finished', 'data_loaded'
129             * @param processorId id of the processor that will be used for this stage
130             * @return builder instance
131             */
132            public Builder add(String intermediate, String completed, String processorId) {
133                if(null == intermediate) throw new IllegalArgumentException("Null intermediate stage provided");
134                if(null == completed) throw new IllegalArgumentException("Null completed stage provided");
135                boolean unique1 = this.stages.add(intermediate);
136                if(!unique1) throw new IllegalArgumentException("Duplicate stage provided: '" + intermediate + "'");
137                boolean unique2 = this.stages.add(completed);
138                if(!unique2) throw new IllegalArgumentException("Duplicate stage provided: '" + completed + "'");
139                this.list.add(new Stage(intermediate, completed, processorId));
140                return this;
141            }
142    
143            /**
144             * @return stage chain instance
145             */
146            public TaskStageChain build() {
147                return new TaskStageChain(list);
148            }
149        }
150    
151        /**
152         * Inner implementation of stage
153         */
154        protected static class Stage implements Serializable {
155            private static final long serialVersionUID = 6127466720110180244L;
156    
157            private final String intermediate;
158            private final String completed;
159            private final String processorId;
160            private final boolean start;
161    
162            Stage(String completed) {
163                this.completed = completed;
164                this.start = true;
165                this.intermediate = null;
166                this.processorId = null;
167            }
168    
169            Stage(String intermediate, String completed, String processorId) {
170                this.intermediate = intermediate;
171                this.completed = completed;
172                this.processorId = processorId;
173                this.start = false;
174            }
175    
176            public String getIntermediate() {
177                if(start) throw new IllegalArgumentException("Start stage: '" + intermediate + "' has no intermediate stage");
178                return intermediate;
179            }
180    
181            public String getCompleted() {
182                return completed;
183            }
184    
185            public String getProcessorId() {
186                if(start)  throw new IllegalArgumentException("Start stage: '" + completed + "' has no processorId");
187                return processorId;
188            }
189    
190            public boolean isStart() {
191                return start;
192            }
193    
194            @Override
195            public boolean equals(Object o) {
196                if (this == o) return true;
197                if (o == null || getClass() != o.getClass()) return false;
198                Stage stage = (Stage) o;
199                return completed.equals(stage.completed);
200            }
201    
202            @Override
203            public int hashCode() {
204                return completed.hashCode();
205            }
206    
207            @Override
208            public String toString() {
209                return completed;
210            }
211        }
212    }
