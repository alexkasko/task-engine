001    package com.alexkasko.tasks;
002    
003    import java.io.Serializable;
004    import java.util.*;
005    
006    /**
007     * Implementation of task stages list. Must be provided by {@link Task} instances. Thread-safe.
008     *
009     * @author alexkasko
010     * Date: 5/22/12
011     * @see TaskEngine
012     * @see Task
013     * @see TaskProcessorProvider
014     */
015    public class TaskStageChain implements Serializable {
016        private static final long serialVersionUID = 7486673100573364684L;
017        protected final Map<String, Stage> stageMap;
018        protected final List<Stage> stageList;
019    
020        /**
021         * Protected constructor for inheritors
022         *
023         * @param stageList list of stages for this chain
024         */
025        protected TaskStageChain(List<Stage> stageList) {
026            if(null == stageList) throw new TaskEngineException("Null stageList provided");
027            this.stageList = stageList;
028            this.stageMap = new LinkedHashMap<String, Stage>();
029            for(Stage ts : stageList) {
030                if(!ts.isStart()) stageMap.put(ts.getIntermediate(), ts);
031                stageMap.put(ts.getCompleted(), ts);
032            }
033        }
034    
035        /**
036         * Finds stage in chain by name. Made public (not package private) for inheritors.
037         *
038         * @param stage stage name
039         * @return {@link Stage} instance for given name
040         */
041        public Stage forName(String stage) {
042            if(null == stage) throw new TaskEngineException("Null stage provided");
043            Stage res = stageMap.get(stage);
044            if(null == res) throw new TaskEngineException(
045                    "Unknown stage provided, valid stages are: [" + stageList + "]");
046            return res;
047        }
048    
049        /**
050         * Gets stage from the chain previous to provided stage. Made public (not package private) for inheritors.
051         *
052         * @param stage input stage
053         * @return previous stage
054         */
055        public Stage previous(Stage stage) {
056            int ind = index(stage);
057            if(0 == ind) throw new TaskEngineException(
058                    "Start stage: [" + stage + "], has no previous stage, valid stages are: [" + stageList + "]");
059            return stageList.get(ind - 1);
060        }
061    
062        /**
063         * Gets stage from the chain next to provided stage. Made public (not package private) for inheritors.
064         *
065         * @param stage input stage
066         * @return next stage
067         */
068        public Stage next(Stage stage) {
069            int ind = index(stage);
070            if(stageList.size() - 1 == ind) throw new TaskEngineException(
071                    "End stage: [" + stage + "], has no next stage, valid stages are: [" + stageList + "]");
072            return stageList.get(ind + 1);
073        }
074    
075        /**
076         * Checks whether provided stage has a next stage. Made public (not package private) for inheritors.
077         *
078         * @param stage input stage
079         * @return whether next stage exists for given stage
080         */
081        public boolean hasNext(Stage stage) {
082            return stageList.size() - 1 > index(stage);
083        }
084    
085        /**
086         * Public method for client-code (so stage names are used instead of stages),
087         * return last completed stage for the given stage.
088         *
089         * @param currentStage current stage
090         * @return given stage name, if it's completed name, completed name of previous stage otherwise.
091         */
092        public String lastCompletedStage(Enum<?> currentStage) {
093            if(null == currentStage) throw new TaskEngineException("Null currentStage provided");
094            return lastCompletedStage(currentStage.name());
095        }
096    
097        /**
098         * Public method for client-code (so stage names are used instead of stages),
099         * return last completed stage for the given stage.
100         *
101         * @param currentStage name of the current stage
102         * @return given stage name, if it's completed name, completed name of previous stage otherwise.
103         */
104        public String lastCompletedStage(String currentStage) {
105            if(null == currentStage) throw new TaskEngineException("Null currentStage provided");
106            Stage stage = forName(currentStage);
107            if(currentStage.equals(stage.getCompleted())) return currentStage;
108            Stage prev = previous(stage);
109            return prev.getCompleted();
110        }
111    
112        /**
113         * Builder instance factory method
114         *
115         * @param startStage start stage
116         * @return {@link Builder} builder for chain
117         */
118        public static Builder builder(Enum<?> startStage) {
119            if(null == startStage) throw new TaskEngineException("Null startStage provided");
120            return builder(startStage.name());
121        }
122    
123        /**
124         * Builder instance factory method
125         *
126         * @param startStage start stage name
127         * @return {@link Builder} builder for chain
128         */
129        public static Builder builder(String startStage) {
130            if(null == startStage) throw new TaskEngineException("Null startStage provided");
131            return new Builder(startStage);
132        }
133    
134        private int index(Stage stage) {
135            if(null == stage) throw new TaskEngineException("Null stage provided");
136            // for short lists this should be faster on list than on set
137            int pos = stageList.indexOf(stage);
138            if(-1 == pos) throw new TaskEngineException("Unknown stage provided, valid stages are: [" + stageList + "]");
139            return pos;
140        }
141    
142        /**
143         * Builder class for {@link TaskStageChain}, not thread-safe
144         */
145        public static class Builder {
146            private final List<Stage> list = new ArrayList<Stage>();
147            private final Set<String> stages = new HashSet<String>();
148    
149            /**
150             * Constructor, protected for inheritors
151             *
152             * @param startStage start stage name
153             */
154            protected Builder(String startStage) {
155                this.list.add(new Stage(startStage));
156                this.stages.add(startStage);
157            }
158    
159            /**
160             * Adds new enum stage to chain
161             *
162             * @param intermediate intermediate stage, e.g. 'running', 'loading_data'
163             * @param completed completed stage, e.g. 'finished', 'data_loaded'
164             * @param processorId id of the processor that will be used for this stage
165             * @return builder instance
166             */
167            public Builder add(Enum<?> intermediate, Enum<?> completed, String processorId) {
168                if(null == intermediate) throw new TaskEngineException("Null intermediate stage provided");
169                if(null == completed) throw new TaskEngineException("Null completed stage provided");
170                return add(intermediate.name(), completed.name(), processorId);
171            }
172    
173            /**
174             * Adds new stage to chain
175             *
176             * @param intermediate intermediate stage name, e.g. 'running', 'loading_data'
177             * @param completed completed stage name, e.g. 'finished', 'data_loaded'
178             * @param processorId id of the processor that will be used for this stage
179             * @return builder instance
180             */
181            public Builder add(String intermediate, String completed, String processorId) {
182                if(null == intermediate) throw new TaskEngineException("Null intermediate stage provided");
183                if(null == completed) throw new TaskEngineException("Null completed stage provided");
184                boolean unique1 = this.stages.add(intermediate);
185                if(!unique1) throw new TaskEngineException("Duplicate stage provided: [" + intermediate + "]");
186                boolean unique2 = this.stages.add(completed);
187                if(!unique2) throw new TaskEngineException("Duplicate stage provided: [" + completed + "]");
188                this.list.add(new Stage(intermediate, completed, processorId));
189                return this;
190            }
191    
192            /**
193             * Creates stage chain instance
194             *
195             * @return stage chain instance
196             */
197            public TaskStageChain build() {
198                return new TaskStageChain(list);
199            }
200        }
201    
202        /**
203         * Inner implementation of stage
204         */
205        protected static class Stage implements Serializable {
206            private static final long serialVersionUID = 6127466720110180244L;
207    
208            protected final String intermediate;
209            protected final String completed;
210            protected final String processorId;
211            protected final boolean start;
212    
213            /**
214             * Constructor for start stage
215             *
216             * @param startStageName name of the start stage
217             */
218            protected Stage(String startStageName) {
219                if(null == startStageName) throw new TaskEngineException("Null startStageName stage provided");
220                this.completed = startStageName;
221                this.start = true;
222                this.intermediate = null;
223                this.processorId = null;
224            }
225    
226            /**
227             * Constructor for stage
228             *
229             * @param intermediate intermediate stage name
230             * @param completed completed stage name
231             * @param processorId processorId for this stage
232             */
233            protected Stage(String intermediate, String completed, String processorId) {
234                if(null == intermediate) throw new TaskEngineException("Null intermediate stage provided");
235                if(null == completed) throw new TaskEngineException("Null completed stage provided");
236                if(null == processorId) throw new TaskEngineException("Null processorId provided");
237                this.intermediate = intermediate;
238                this.completed = completed;
239                this.processorId = processorId;
240                this.start = false;
241            }
242    
243            /**
244             * Returns intermediate name of this stage
245             *
246             * @return intermediate name of this stage
247             */
248            public String getIntermediate() {
249                if(start) throw new TaskEngineException("Start stage: [" + intermediate + "] has no intermediate stage");
250                return intermediate;
251            }
252    
253            /**
254             * Returns completed name of this stage
255             *
256             * @return completed name of this stage
257             */
258            public String getCompleted() {
259                return completed;
260            }
261    
262            /**
263             * Returns processor ID for this stage
264             *
265             * @return processor ID for this stage
266             */
267            public String getProcessorId() {
268                if(start)  throw new TaskEngineException("Start stage: [" + completed + "] has no processorId");
269                return processorId;
270            }
271    
272            /**
273             * Whether this stage is start stage
274             *
275             * @return whether this stage is start stage
276             */
277            public boolean isStart() {
278                return start;
279            }
280    
281            /**
282             * {@inheritDoc}
283             */
284            @Override
285            public boolean equals(Object o) {
286                if (this == o) return true;
287                if (o == null || getClass() != o.getClass()) return false;
288                Stage stage = (Stage) o;
289                return completed.equals(stage.completed);
290            }
291    
292            /**
293             * {@inheritDoc}
294             */
295            @Override
296            public int hashCode() {
297                return completed.hashCode();
298            }
299    
300            /**
301             * {@inheritDoc}
302             */
303            @Override
304            public String toString() {
305                return completed;
306            }
307        }
308    }
