001    package com.alexkasko.tasks;
002    
003    /**
004     * Interface for asynchronous multistage tasks.
005     * It's implied, that task has two different fields to represent current state: "stage" and "status".
006     * Stage (e.g. 'loading_data', 'finished') is and task processing state,
007     * Status (e.g. 'error', 'processing') is a task application status
008     *
009     * @author alexkasko
010     * Date: 5/17/12
011     * @see TaskEngine
012     * @see TaskStageChain
013     * @see TaskManager
014     */
015    public interface Task {
016        /**
017         * Should be implemented as static method on the upper level of hierarchy
018         *
019         * @return all stages this task may be into during processing
020         */
021        TaskStageChain stageChain();
022    
023        /**
024         * Returns task instance unique id
025         *
026         * @return task instance unique id
027         */
028        long getId();
029    
030        /**
031         * Current stage for this task
032         *
033         * @return current stage for this task
034         */
035        String getStageName();
036    }
