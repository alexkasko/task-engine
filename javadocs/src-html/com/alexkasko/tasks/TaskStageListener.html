001    package com.alexkasko.tasks;
002    
003    /**
004     * Additional operations that will be executed before/after task stage processing.
005     * Listeners are separated from task processors to (among others)
006     * simplify their declarative transaction management.
007     *
008     * @author alexkasko
009     * Date: 6/30/12
010     * @see TaskStageProcessor
011     * @see TaskEngine
012     */
013    public interface TaskStageListener {
014        /**
015         * Will be called by {@link TaskEngine} before/after stage processing
016         *
017         * @param taskId task ID
018         */
019        void fire(long taskId);
020    }
